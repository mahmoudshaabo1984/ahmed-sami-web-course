/* تكبير العنوان الأول */
.myclass {
  color: #ff0000; /* أحمر (من الدرس السابق) */

  /* هنا الدرس الجديد: حجم خط كبير جداً */
  font-size: 50px;
}

/* تنسيق العنوان الثاني */
#myid {
  color: #0000ff; /* أزرق */

  /* حجم خط أصغر قليلاً من الأول */
  font-size: 30px;
}

/* تنسيق الفقرات (لجعل القراءة أسهل) */
.paragraph-text {
  /* الحجم الافتراضي في المتصفح هو 16px، نحن سنكبره لـ 20 */
  font-size: 20px;

  /* لون رمادي غامق مريح للعين (ذوق الأستاذ) */
  color: #333;
}

/* 1. العنوان الأول */
.myclass {
  color: #ff0000; /* لون أحمر (Hex) */
  
  /* الإضافة الجديدة: تكبير الخط */
  font-size: 50px;
}

/* 2. العنوان الثاني */
#myid {
  color: #0f0; /* لون أخضر (Hex مختصر) */
  
  /* الإضافة الجديدة: حجم أصغر قليلاً */
  font-size: 30px;
}

/* 3. الفقرة الأولى */
.p1 {
  color: #0000ff; /* لون أزرق */
  font-weight: bold;
  
  /* تعديل الحجم */
  font-size: 25px;
}

/* 4. الفقرة الثانية */
.p2 {
  color: #000; /* لون أسود */
  
  /* تعديل الحجم */
  font-size: 20px;
}
