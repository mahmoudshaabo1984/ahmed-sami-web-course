------------------------------------------------------------
الكود الأول - خاطئ (position relative مفقودة من الحاوية)
------------------------------------------------------------

index.html:
-----------
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>home</title>
    <link rel="stylesheet" href="main.css" />
  </head>
  <body>
    <div class="container">
      <h1>start</h1>
      <h1>end</h1>
    </div>
  </body>
</html>

main.css:
---------
.container {
  width: 600px;            /* عرض الحاوية */
  height: 500px;           /* ارتفاع الحاوية */
  border: 3px solid black; /* إطار أسود سميك */
  margin: 10px auto;       /* توسيط الحاوية أفقياً */
  /* المشكلة: لا يوجد position: relative هنا */
  /* لذلك h1 المطلق سيتموضع حسب نافذة المتصفح لا الحاوية */
}

.container h1 {
  position: absolute; /* يخرج من التدفق الطبيعي ويتموضع بشكل مطلق */
  top: 30px;          /* مسافة 30px من أعلى أقرب عنصر أب له position */
}


------------------------------------------------------------
الكود الثاني - المصحح (إضافة position relative للحاوية)
------------------------------------------------------------

index.html: (نفس الكود السابق)

main.css:
---------
.container {
  width: 600px;            /* عرض الحاوية */
  height: 500px;           /* ارتفاع الحاوية */
  border: 3px solid black; /* إطار أسود سميك */
  margin: 10px auto;       /* توسيط الحاوية أفقياً */
  position: relative;      /* تحديد الحاوية كمرجع لـ absolute بداخلها */
}

.container h1 {
  position: absolute; /* يتموضع حسب الحاوية التي لها position: relative */
  top: 30px;          /* مسافة 30px من أعلى الحاوية */
}


------------------------------------------------------------
الكود الثالث - إضافة z-index
------------------------------------------------------------

index.html: (نفس الكود السابق)

main.css:
---------
.container {
  width: 600px;            /* عرض الحاوية */
  height: 500px;           /* ارتفاع الحاوية */
  border: 3px solid black; /* إطار أسود سميك */
  margin: 10px auto;       /* توسيط الحاوية أفقياً */
  position: relative;      /* مرجع لأي عنصر absolute بداخلها */
}

.container h1 {
  position: absolute; /* يتموضع حسب الحاوية */
  top: 40px;          /* مسافة 40px من أعلى الحاوية */
  z-index: 10;        /* يظهر فوق العناصر الأخرى (الرقم الأعلى = الأمام) */
}


------------------------------------------------------------
الكود الرابع والنهائي - تجربة position fixed مع body طويل
------------------------------------------------------------

index.html: (نفس الكود السابق)

main.css:
---------
body {
  height: 4000px; /* جعل الصفحة طويلة لتجربة التمرير مع fixed */
}

.container {
  width: 600px;            /* عرض الحاوية */
  height: 500px;           /* ارتفاع الحاوية */
  border: 3px solid black; /* إطار أسود سميك */
  margin: 0 auto;          /* توسيط الحاوية أفقياً */
  position: relative;      /* مرجع لأي عنصر absolute بداخلها */
}

.container h1 {
  position: fixed; /* يثبت في مكانه على الشاشة حتى عند التمرير */
  z-index: 10;     /* يظهر فوق كل العناصر الأخرى */
}


============================================================
ملاحظات مهمة عن أنواع position الستة:
============================================================

1. static   - الافتراضي، لا يتأثر بـ top/left/right/bottom
2. relative - يتحرك من مكانه الأصلي، ويكون مرجعاً لـ absolute
3. absolute - يتموضع حسب أقرب أب له position (غير static)
4. fixed    - يثبت على الشاشة بغض النظر عن التمرير
5. initial  - يعيد القيمة للوضع الافتراضي (static)
6. inherit  - يرث قيمة position من العنصر الأب

القاعدة: عند استخدام absolute داخل عنصر، يجب إضافة
position: relative للعنصر الأب حتى يكون هو المرجع.
