<!doctype html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>List Styling Lesson</title>
    <link rel="stylesheet" href="style.css" />
  </head>
  <body>
    <ul class="nav">
      <li><a href="#">home</a></li>
      <li><a href="#">contact</a></li>
      <li><a href="#">shop</a></li>
      <li><a href="#">about</a></li>
    </ul>

    <h3>shop items</h3>
    <ol class="shop-item">
      <li>item1</li>
      <label for="item2"></label>
      <li>item3</li>
      <ul>
        <li><a href="#">home</a></li>
        <li><a href="#">contact</a></li>
        <li><a href="#">shop</a></li>
        <li><a href="#">about</a></li>
      </ul>
      <label for="item4"></label>
      <li>item5</li>
    </ol>
  </body>
</html>

--------------------------------------------------------------------------------

.nav {

}

.nav li {
  display: inline-block;
  padding: 5px;
  margin: 0 5px;
}

.nav li a {
  text-decoration: none;
  color: #000;
}

.nav li a:hover {
  font-weight: bold;
  color: #fff;
}
