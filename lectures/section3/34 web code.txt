/* الرابط في حالته الطبيعية */
a {
  text-decoration: none;   /* إزالة الخط السفلي */
  background-color: #ff0;  /* خلفية صفراء */
  border: 2px solid #000;  /* إطار أسود */
  color: #000;             /* نص أسود */
  border-radius: 3%;       /* حواف دائرية خفيفة */
}

/* عند تمرير الماوس فوق الرابط */
a:hover {
  background: #333;        /* خلفية رمادية غامقة */
  color: #fff;             /* نص أبيض */
  font-weight: 900;        /* خط سميك جداً */
}

/* الرابط بعد الزيارة */
a:visited {
  color: #0c0;             /* لون أخضر يدل على الزيارة السابقة */
}
